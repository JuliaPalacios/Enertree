trees_beast_ess <- read.nexus("100THOUSANDTREES-sequences.trees")
beast_ess_fmats <- lapply(trees_beast_ess, gen_Fmat, tol = 8)

beast_uniform_mds_sub_samp <- min(1000, length(beast_ess_fmats))
set.seed(52)
beast_uniform_idx <- sample(length(beast_ess_fmats), beast_uniform_mds_sub_samp)
beast_uniform_beast_fmats <- beast_ess_fmats[beast_uniform_idx]

uniform_ess_encodings <- rEncod(
  m = beast_uniform_mds_sub_samp,
  n = n_tips,
  distr = "uniform"
)
uniform_ess_fmats <- lapply(uniform_ess_encodings, Fmat_from_myencod)

beast_uniform_mds_fmats <- c(beast_uniform_beast_fmats, uniform_ess_fmats)
beast_uniform_mds_labels <- c(
  rep("BEAST", length(beast_uniform_beast_fmats)),
  rep("Uniform", length(uniform_ess_fmats))
)

beast_uniform_mds_colors <- c("BEAST" = "steelblue", "Uniform" = "tomato")

beast_uniform_mds_D <- matrix(0, length(beast_uniform_mds_fmats), length(beast_uniform_mds_fmats))
for (i in 2:length(beast_uniform_mds_fmats)) {
  for (j in 1:(i - 1)) {
    beast_uniform_mds_D[i, j] <- distance_Fmat(
      beast_uniform_mds_fmats[[i]],
      beast_uniform_mds_fmats[[j]],
      dist = "l2"
    )^2
    beast_uniform_mds_D[j, i] <- beast_uniform_mds_D[i, j]
  }
}

beast_uniform_mds <- cmdscale(as.dist(beast_uniform_mds_D), k = 2, eig = TRUE)
beast_uniform_coords <- as.data.frame(beast_uniform_mds$points)
colnames(beast_uniform_coords) <- c("Dim1", "Dim2")
beast_uniform_coords$group <- factor(beast_uniform_mds_labels, levels = c("BEAST", "Uniform"))

p_beast_uniform <- ggplot(beast_uniform_coords, aes(x = Dim1, y = Dim2, color = group)) +
  geom_point(alpha = 0.7, size = 1.8) +
  scale_color_manual(values = beast_uniform_mds_colors) +
  labs(
    title = "BEAST Trees vs Uniform F Matrices",
    x = "MDS 1",
